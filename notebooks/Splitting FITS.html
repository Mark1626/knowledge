import matplotlib.pyplot as plt
from astropy.io import fits
from astropy.wcs import WCS

def plot_fits(hdu):
    wcs = WCS(hdu.header, naxis=2)
    plt.subplots(figsize=(6, 6))
    plt.subplots_adjust(bottom=0.25, top=0.9, right=0.7)
    ax = plt.subplot(211, projection=wcs)

    im = ax.imshow(hdu.data[0,0,:,:], origin='lower', cmap='viridis', vmin=0, vmax=300e-6)
    ax.set_xlabel('RA')
    ax.set_ylabel('Dec')
    ax.set_autoscale_on(False)
    ax.set_title(file)
    plt.colorbar(im, ax=ax)

    plt.show()

from astropy import wcs
from astropy import units as u

# New header from WCS
w = wcs.WCS(naxis=4)

pixel_size = 2.5 * u.arcsec
pixel_size_deg = pixel_size.to(u.deg)

w.wcs.crpix = [1024, 1024, 1, 1]
w.wcs.cunit = ['deg', 'deg', 'Hz', '']
w.wcs.cdelt = np.array([-pixel_size_deg.value, pixel_size_deg.value, 1, 1])
w.wcs.crval = [1.260899000000E+02, -6.814048888889E+01, 1, 1]
w.wcs.ctype = ["RA---SIN", "DEC--SIN", 'FREQ', 'STOKES']

header = w.to_header()

hdu_split = fits.PrimaryHDU(data=data_split, header=header)
